import requests

def download_media(telegram_link):
    try:
        response = requests.get(telegram_link)
        response.raise_for_status()  # Raises an HTTPError for bad responses

        with open('downloaded_media', 'wb') as file:

        print("Media downloaded successfully.")

### PHP
This example demonstrates how to use cURL to download media from a Telegram link in PHP.

php
<?php
function downloadMedia($telegramLink) {
    $ch = curl_init($telegramLink);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);

    $response = curl_exec($ch);

    if (curl_errno($ch)) {
        echo 'cURL Error: ' . curl_error($ch);
    } else {
        file_put_contents('downloaded_media', $response);
        echo "Media downloaded successfully.";
    }

    curl_close($ch);
}

$telegramLink = "https://hdstockimages.com/telegram-downloader-2/";
downloadMedia($telegramLink);
?>

async function downloadMedia(telegramLink) {
    try {
        const response = await fetch(telegramLink);
        
        if (!response.ok) {
            throw new Error(`HTTP error! Status: ${response.status}`);
        }

        const blob = await response.blob();
        const url = window.URL.createObjectURL(blob);
        const link = document.createElement('a');
        link.href = url;
        link.setAttribute('download', 'downloaded_media');
        document.body.appendChild(link);
        link.click();
        link.remove();

        console.log("Media downloaded successfully.");

    } catch (error) {
        console.error("An error occurred:", error);
    }
}
